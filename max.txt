TECHNICAL IMPLEMENTATION DETAILS

REAL-TIME TRACKING IMPLEMENTATION:
```javascript
// Location tracking with 1.5-second intervals
trackingInterval.value = window.setInterval(() => {
  navigator.geolocation.getCurrentPosition(
    (position) => {
      const location: UserLocation = {
        lat: position.coords.latitude,
        lng: position.coords.longitude,
        accuracy: position.coords.accuracy,
        timestamp: new Date(position.timestamp)
      }
      userLocation.value = location
      lastUpdateTime.value = new Date()
      
      // Update route distance if destination is set
      if (destinationStation.value) {
        routeDistance.value = calculateDistance(location, destinationStation.value)
      }
    },
    (error) => {
      console.error('Tracking error:', error)
      if (error.code === error.PERMISSION_DENIED) {
        stopRealTimeTracking()
      }
    },
    { enableHighAccuracy: true, timeout: 5000, maximumAge: 0 }
  )
}, 1500) // 1.5 seconds
```

DISTANCE CALCULATION:
```javascript
// Haversine formula for accurate distance
const calculateDistance = (from: UserLocation, to: Station): number => {
  const R = 6371 // Earth's radius in kilometers
  const dLat = (to.lat - from.lat) * Math.PI / 180
  const dLng = (to.lng - from.lng) * Math.PI / 180
  const a = Math.sin(dLat/2) * Math.sin(dLat/2) +
            Math.cos(from.lat * Math.PI / 180) * Math.cos(to.lat * Math.PI / 180) *
            Math.sin(dLng/2) * Math.sin(dLng/2)
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a))
  return R * c
}
```

const searchStations = (query: string): Station[] => {
  const searchTerm = query.toLowerCase()
  return stations.filter(station => 
    station.name.toLowerCase().includes(searchTerm) ||
    station.area?.toLowerCase().includes(searchTerm) ||
    station.description?.toLowerCase().includes(searchTerm)
  )
}
